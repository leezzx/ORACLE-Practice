-- SUBSTR(문자열, 시작위치, 길이) : 문자열 추출 함수

SELECT * FROM MEMBER WHERE SUBSTR(BIRTHDAY, 6, 2) NOT IN ('07', '08, '09');
SELECT * FROM MEMBER WHERE PHONE IS NULL AND SUBSTR(BIRTHDAY, 6, 2) IN ('07', '08, '09');



-- CONCAT( , ) : 문자열 덧샘 함수

SELECT CONCAT('홍', '길동') FROM DUAL;



-- TRIM() : 문자열 공백 삭제 LTRIM, RTRIM

SELECT LTRIM('    HELLO    ') FROM DUAL;
SELECT RTRIM('    HELLO    ') FROM DUAL;
SELECT TRIM('    HELLO    ') FROM DUAL;



-- LOWER() : 소문자로, UPPER() : 대문자로

SELECT LOWER('NewlEC') FROM DUAL;
SELECT UPPER('nEWLec') FROM DUAL;
SELECT * FROM NAME WHERE UPPER(ID) = 'newlec'; -- 대소문자 가리지 않는 경우



-- REPLACE : 해당 단어 교체, TRANSLATE : 해당 문자열의 모든 요소를 교체

SELECT REPLACE('WHERE WE ARE', 'WE', 'YOU') FROM DUAL; -- WHERE YOU ARE
SELECT TRANSLATE('WHERE WH ARE', 'WE', 'YOU') FROM DUAL; -- YHORO YH ARO



-- INSTR(문자열, 검색문자열, 위치, 찾을 수)

SELECT INSTR('ALL WE NEED TO IS JUST TO...', 'TO') FROM DUAL; -- 13
SELECT INSTR('ALL WE NEED TO IS JUST TO...', 'TO', 15) FROM DUAL; -- 24
SELECT INSTR('ALL WE NEED TO IS JUST TO...', 'TO', 1, 2) FROM DUAL; -- 24



-- LENGTH : 문자열 길이를 반환

SELECT LENGTH('SKDKD') FROM DUAL;



-- ASCII CODE

SELECT ASCII('A') FROM DUAL;
SELECT CHR('65') FROM DUAL;
